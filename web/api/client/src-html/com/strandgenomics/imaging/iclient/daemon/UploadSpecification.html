001    package com.strandgenomics.imaging.iclient.daemon;
002    
003    import java.io.Serializable;
004    
005    import com.strandgenomics.imaging.iclient.local.RawExperiment;
006    import com.strandgenomics.imaging.iclient.local.UploadStatus;
007    
008    /**
009     * The specification of the record, original data and user data to be uploaded
010     * 
011     * @author Anup Kulkarni
012     */
013    public class UploadSpecification implements Serializable{
014            /**
015             * 
016             */
017            private static final long serialVersionUID = 2944488786548908567L;
018            /**
019             * details of user session
020             */
021            private UserSessionSpecification session;
022            /**
023             * experiment
024             */
025            private RawExperiment experiment;
026            /**
027             * upload status
028             */
029            private UploadStatus status;
030            /**
031             * upload identifier given/used by the upload daemon
032             */
033            private long uploadId;
034            /**
035             * path where the object is stored to maintain session of the upload daemon
036             */
037            private String filepath;
038            /**
039             * true if the upload is canceled
040             */
041            private boolean canceled;
042            /**
043             * message set by the upload daemon
044             */
045            private String message = "";
046    
047            public UploadSpecification(UserSessionSpecification session, RawExperiment experiment) 
048            {
049                    this.session = session;
050                    this.experiment = experiment;
051                    this.status = UploadStatus.QueuedBackground;
052            }
053            /**
054             * sets the upload id
055             * @param uploadId
056             */
057            void setUploadId(long uploadId)
058            {
059                    this.uploadId = uploadId;
060            }
061            /**
062             * 
063             * @return upload identifier for the upload spec
064             */
065            public long getUploadId()
066            {
067                    return this.uploadId;
068            }
069            
070            /**
071             * sets the status of this upload specification
072             * @param status
073             */
074            public void setStatus(UploadStatus status)
075            {
076                    this.status = status;
077            }
078            /**
079             * upload status
080             * @return
081             */
082            public UploadStatus getStatus()
083            {
084                    return this.status;
085            }
086            /**
087             * returns the message
088             * @return
089             */
090            public String getMessage()
091            {
092                    return this.message;
093            }
094            
095            void setMessage(String string)
096            {
097                    this.message = string;
098            }
099            
100            void setCanceled()
101            {
102                    this.canceled = true;
103            }
104            
105            boolean isCanceled()
106            {
107                    return this.canceled;
108            }
109            
110            /**
111             * returns location where is spec is stored
112             * @return
113             */
114            String getSessionFilepath()
115            {
116                    return this.filepath;
117            }
118            
119            void setSessionFilepath(String path)
120            {
121                    this.filepath = path;
122            }
123    
124            public RawExperiment getExperiment()
125            {
126                    return experiment;
127            }
128            
129            public String getProjectName()
130            {
131                    return session.projectName;
132            }
133    
134            public String getUserName()
135            {
136                    return session.userLogin;
137            }
138    
139            public String getAccessKey()
140            {
141                    return session.accessKey;
142            }
143    
144            public String getHost()
145            {
146                    return session.host;
147            }
148    
149            public int getPort()
150            {
151                    return session.port;
152            }
153            
154            public boolean isSSL()
155            {
156                    return session.useSSL;
157            }
158    }
