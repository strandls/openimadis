041    import java.io.BufferedReader;
042    import java.io.File;
043    import java.io.FileInputStream;
044    import java.io.FileOutputStream;
045    import java.io.InputStream;
046    import java.io.InputStreamReader;
047    import java.io.OutputStream;
048    import java.security.KeyStore;
049    import java.security.MessageDigest;
050    import java.security.cert.CertificateException;
051    import java.security.cert.X509Certificate;
052    
053    import javax.net.ssl.SSLContext;
054    import javax.net.ssl.SSLException;
055    import javax.net.ssl.SSLSocket;
056    import javax.net.ssl.SSLSocketFactory;
057    import javax.net.ssl.TrustManager;
058    import javax.net.ssl.TrustManagerFactory;
059    import javax.net.ssl.X509TrustManager;
060    
061    /**
062     * Class used to add the server's certificate to the KeyStore
063     * with your trusted certificates.
064     */
065    public class InstallCert {
066    
067        public static void main(String[] args) throws Exception {
068            String host;
069            int port;
070            char[] passphrase;
071            if ((args.length == 1) || (args.length == 2)) {
072                String[] c = args[0].split(":");
073                host = c[0];
074                port = (c.length == 1) ? 443 : Integer.parseInt(c[1]);
075                String p = (args.length == 1) ? "changeit" : args[1];
076                passphrase = p.toCharArray();
077            } else {
078                System.out.println("Usage: java InstallCert <host>[:port] [passphrase]");
079                return;
080            }
081    
082            File file = new File("jssecacerts");
083            if (file.isFile() == false) {
084                char SEP = File.separatorChar;
085                File dir = new File(System.getProperty("java.home") + SEP
086                        + "lib" + SEP + "security");
087                file = new File(dir, "jssecacerts");
088                if (file.isFile() == false) {
089                    file = new File(dir, "cacerts");
090                }
091            }
092            System.out.println("Loading KeyStore " + file + "...");
093            InputStream in = new FileInputStream(file);
094            KeyStore ks = KeyStore.getInstance(KeyStore.getDefaultType());
095            ks.load(in, passphrase);
096            in.close();
097    
098            SSLContext context = SSLContext.getInstance("TLS");
099            TrustManagerFactory tmf =
100                    TrustManagerFactory.getInstance(TrustManagerFactory.getDefaultAlgorithm());
101            tmf.init(ks);
102            X509TrustManager defaultTrustManager = (X509TrustManager) tmf.getTrustManagers()[0];
103            SavingTrustManager tm = new SavingTrustManager(defaultTrustManager);
104            context.init(null, new TrustManager[]{tm}, null);
105            SSLSocketFactory factory = context.getSocketFactory();
106    
107            System.out.println("Opening connection to " + host + ":" + port + "...");
108            SSLSocket socket = (SSLSocket) factory.createSocket(host, port);
109            socket.setSoTimeout(10000);
110            try {
111                System.out.println("Starting SSL handshake...");
112                socket.startHandshake();
113                socket.close();
114                System.out.println();
115                System.out.println("No errors, certificate is already trusted");
116            } catch (SSLException e) {
117                System.out.println();
118                e.printStackTrace(System.out);
119            }
120    
121            X509Certificate[] chain = tm.chain;
122            if (chain == null) {
123                System.out.println("Could not obtain server certificate chain");
124                return;
125            }
126    
127            BufferedReader reader =
128                    new BufferedReader(new InputStreamReader(System.in));
129    
130            System.out.println();
131            System.out.println("Server sent " + chain.length + " certificate(s):");
132            System.out.println();
133            MessageDigest sha1 = MessageDigest.getInstance("SHA1");
134            MessageDigest md5 = MessageDigest.getInstance("MD5");
135            for (int i = 0; i < chain.length; i++) {
136                X509Certificate cert = chain[i];
137                System.out.println
138                        (" " + (i + 1) + " Subject " + cert.getSubjectDN());
139                System.out.println("   Issuer  " + cert.getIssuerDN());
140                sha1.update(cert.getEncoded());
141                System.out.println("   sha1    " + toHexString(sha1.digest()));
142                md5.update(cert.getEncoded());
143                System.out.println("   md5     " + toHexString(md5.digest()));
144                System.out.println();
145            }
146    
147            System.out.println("Enter certificate to add to trusted keystore or 'q' to quit: [1]");
148            String line = reader.readLine().trim();
149            int k;
150            try {
151                k = (line.length() == 0) ? 0 : Integer.parseInt(line) - 1;
152            } catch (NumberFormatException e) {
153                System.out.println("KeyStore not changed");
154                return;
155            }
156    
157            X509Certificate cert = chain[k];
158            String alias = host + "-" + (k + 1);
159            ks.setCertificateEntry(alias, cert);
160    
161            OutputStream out = new FileOutputStream("jssecacerts");
162            ks.store(out, passphrase);
163            out.close();
164    
165            System.out.println();
166            System.out.println(cert);
167            System.out.println();
168            System.out.println
169                    ("Added certificate to keystore 'jssecacerts' using alias '"
170                            + alias + "'");
171        }
172    
173        private static final char[] HEXDIGITS = "0123456789abcdef".toCharArray();
174    
175        private static String toHexString(byte[] bytes) {
176            StringBuilder sb = new StringBuilder(bytes.length * 3);
177            for (int b : bytes) {
178                b &= 0xff;
179                sb.append(HEXDIGITS[b >> 4]);
180                sb.append(HEXDIGITS[b & 15]);
181                sb.append(' ');
182            }
183            return sb.toString();
184        }
185    
186        private static class SavingTrustManager implements X509TrustManager {
187    
188            private final X509TrustManager tm;
189            private X509Certificate[] chain;
190    
191            SavingTrustManager(X509TrustManager tm) {
192                this.tm = tm;
193            }
194    
195            public X509Certificate[] getAcceptedIssuers() {
196                throw new UnsupportedOperationException();
197            }
198    
199            public void checkClientTrusted(X509Certificate[] chain, String authType)
200                    throws CertificateException {
201                throw new UnsupportedOperationException();
202            }
203    
204            public void checkServerTrusted(X509Certificate[] chain, String authType)
205                    throws CertificateException {
206                this.chain = chain;
207                tm.checkServerTrusted(chain, authType);
208            }
209        }
