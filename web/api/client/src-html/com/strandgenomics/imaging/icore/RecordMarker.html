001    package com.strandgenomics.imaging.icore;
002    
003    /**
004     * the status of the record: Active/Deleted/Archived etc
005     * 
006     * @author Anup Kulkarni
007     */
008    public enum RecordMarker {
009    
010            /**
011             * record is active in the system
012             */
013            Active {
014                    public String toString()
015                    {
016                            return "Active";
017                    }
018            },
019            /**
020             * record is deleted from the system
021             */
022            Deleted {
023                    public String toString()
024                    {
025                            return "Deleted";
026                    }
027            },
028            /**
029             * record is archived
030             */
031            Archived {
032                    public String toString()
033                    {
034                            return "Archived";
035                    }
036            },
037            /**
038             * record is under construction
039             */
040            UNDER_CONSTRUCTION {
041                    public String toString()
042                    {
043                            return "Under Construction";
044                    }
045            }
046    }
