020    package com.strandgenomics.imaging.icore;
021    
022    import java.io.Serializable;
023    
024    /**
025     * Represents a site of a record. By default, the records are single site records till they are merged
026     * as a multi-site record
027     * @author nimisha
028     */
029    public class Site implements Serializable {
030    
031            private static final long serialVersionUID = 5116314082882541146L;
032            
033            /**
034             * name of the site
035             */
036            protected String name;
037            /**
038             * a site is mapped to the series number of a record
039             */
040            protected final int seriesNo;
041    
042            /**
043             * creates a default site 
044             * @param name name of this site
045             */
046            public Site(int seriesNo, String name)
047            {
048                    this.seriesNo = seriesNo;
049                    this.name = name;
050            }
051            
052        /**
053         * A record may be part of a series (of records). 
054         * This method returns the series number for this record if any. default is zero
055         * @return the series number for this record if any zero otherwise
056         */
057        public int getSeriesNo()
058        {
059            return seriesNo;
060        }
061    
062        /**
063         * series name
064         * @return name of the series
065         */
066            public String getName() {
067                    return name;
068            }
069    
070            
071            public void setName(String name) 
072            {
073                    this.name = name;
074            }
075            
076        @Override
077        public int hashCode()
078        {
079            return seriesNo;
080        }
081        
082        public boolean equals(Object obj)
083        {
084            if(obj != null && obj instanceof Site)
085            {
086                    Site that = (Site) obj;
087                    if(this == that) return true;
088                    
089                    return this.seriesNo == that.seriesNo;
090            }
091            return false;
092        }
093    }
