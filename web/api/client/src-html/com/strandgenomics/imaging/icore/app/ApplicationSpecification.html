021    package com.strandgenomics.imaging.icore.app;
022    
023    import java.util.Set;
024    
025    /**
026     * Detail specification of an application
027     * @author arunabha
028     *
029     */
030    public class ApplicationSpecification extends Application {
031            
032            private static final long serialVersionUID = -17932380155310894L;
033    
034            /**
035             * broad area (or domain) where this application is relevant - for example Quality Control, Statistics 
036             */
037            public final String categoryName;
038            
039            /**
040             * general description of the application
041             */
042            private String description;
043    
044            /**
045             * list of parameters
046             */
047            public Set<Parameter> parameters;
048            
049            /**
050             * server side unique identifier for application
051             */
052            public final String clientId;
053            
054            /**
055             * Defines a application with its name, version, category name, description, parameters and constraints
056             * @param name
057             * @param version
058             * @param categoryName
059             * @param description
060             * @param parameters
061             * @param publisherKey
062             */
063        public ApplicationSpecification(String clientId, String name, String version, String categoryName, String description, Set<Parameter> parameters)
064        {
065            super(name, version);
066            
067            this.description  = description;
068            this.categoryName = categoryName;
069            this.parameters   = parameters;
070            this.clientId = clientId;
071        }
072        
073        /**
074         * updates the description to specified one
075         * @param description
076         */
077        public void setDescription(String description)
078        {
079            this.description = description;
080        }
081        
082            /**
083             * Returns a short description of this application
084             * @return a short description of this application
085             */
086            public String getDescription()
087            {
088                    return description;
089            }
090            
091            /**
092             * Returns server side unique identifier for application
093             * @return client id
094             */
095            public String getID()
096            {
097                    return clientId;
098            }
099            
100            /**
101             * Returns the category name 
102             * @return
103             */
104            public String getCategory()
105            {
106                    return categoryName;
107            }
108            
109            /**
110             * Returns a general description of each parameter needed for application execution
111             * @return a general description of each parameter needed for application execution
112             */
113            public Set<Parameter> getParameters()
114            {
115                    return parameters;
116            }
117    }
